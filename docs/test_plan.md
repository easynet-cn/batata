  Phase 3: Property-Based & Fuzz Testing (Priority: Medium)                                                                                                                                                                                                                     
                                                                                                                                                                                                                                                                                
  3.1 Property-Based Tests (proptest)
                                                                                                                                                                                                                                                                                
  // Example: config key validation
  proptest! {                                                                                                                                                                                                                                                                   
      #[test]                                                                                                                                                                                                                                                                   
      fn config_dataId_roundtrip(data_id in "[a-zA-Z0-9._-]{1,256}") {                                                                                                                                                                                                          
          let published = publish_config(&data_id, "group", "content");                                                                                                                                                                                                         
          let queried = query_config(&data_id, "group");                                                                                                                                                                                                                        
          prop_assert_eq!(published.content, queried.content);                                                                                                                                                                                                                  
      }                                                                                                                                                                                                                                                                         
                                                                                                                                                                                                                                                                                
      #[test]                                                                                                                                                                                                                                                                   
      fn naming_instance_metadata_arbitrary(
          meta in prop::collection::hash_map("[a-z]{1,10}", "[a-z]{1,50}", 0..20)                                                                                                                                                                                               
      ) {                                                                                                                                                                                                                                                                       
          let instance = register_with_metadata(meta.clone());                                                                                                                                                                                                                  
          let queried = query_instance(instance.id);                                                                                                                                                                                                                            
          prop_assert_eq!(queried.metadata, meta);                                                                                                                                                                                                                              
      }                                                                                                                                                                                                                                                                         
  }                                                                                                                                                                                                                                                                             
                                                                                                                                                                                                                                                                                
  Target areas:                                                                                                                                                                                                                                                                 
  - Config data_id / group / namespace validation boundaries
